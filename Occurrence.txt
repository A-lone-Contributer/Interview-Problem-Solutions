# Count the number of 2s as digit in all numbers from 0 to n.

# Asked in amazon
# Difficulty -> Easy

# Below is the concise python implementation


def occurrenceof2(n):
    # count variable to count the occurrence of 2's
    count = 0
    # traversing till the target
    for i in range(1, n + 1):
        # counting the nbr of 2's by casting each number to string
        count += str(i).count('2')

    # return count
    return count


# Test cases
print(occurrenceof2(22))
print(occurrenceof2(100))

# Time complexity : O(n)
# Space complexity : O(1)
